    --attribute-definitions \
        AttributeName=roomId,AttributeType=S \

    --key-schema \
        AttributeName=roomId,KeyType=HASH \
        AttributeName=slotId,KeyType=RANGE \
    --billing-mode PAY_PER_REQUEST
